# install

> **npm install firedb.js**


# example




```js

const db = require("firedb.js");

db.add("fire.add",+1)
db.subtract("fire.add",-1)
db.set("fire.okey", 22);
db.push("fire.array", 32);
db.get("fire"); // {"okey": 22, "array": [32]}
```
